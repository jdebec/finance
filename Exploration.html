def moving_average(serie, n=20) :
    ret = np.cumsum(serie, dtype=float)
    ret[n:] = ret[n:] - ret[:-n]
    return ret[n - 1:] / n

Simulation of automated sell/buy strategy

  • If long MA crosses upper short MA => buy
  • If long MA crosses under short MA => sell
  • Fee on trades: 1%
  • Starts with capital not invested

def simulation_stategy():
    pf = [1]
    invested = False
    buy_date = []
    sell_date =[]
    
    for index in range(len(calendar)):
        
        # update portfolio value
        if invested:
            change_mv = prices[index] / prices[index - 1]
            pf.append(pf[-1] * change_mv)
        
        else:
            pf.append(pf[-1])
            
        # buy/sell decision? => if MA crosses each other
        if index in co_dates:
            
            MAS_param = 5
            MAL_param = 100
            MAL_value = MAL[index - MAL_param -1]
            MAS_value = MAS[index - MAS_param -1]
                
            # buy
            if (MAL_value <  MAS_value and invested == False):
                pf[-1] = 0.99 * pf[-1]
                invested = True
                buy_date.append(index)
                
            # sell
            if (MAL_value >  MAS_value and invested == True):
                pf[-1] = 0.99 * pf[-1]
                invested = False
                sell_date.append(index)
    pf = np.asarray(pf)         
    return (pf[1:], buy_date, sell_date)
